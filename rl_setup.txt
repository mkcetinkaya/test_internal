#Nvidia-driver 470 kurulumu
apt-get update && apt-get install -y linux-headers-$(uname -r) && apt-get install -y --no-install-recommends nvidia-driver-470

#İlgili repository’lerin eklenmesi
distribution=$(. /etc/os-release;echo $ID$VERSION_ID) && curl -s -L https://nvidia.github.io/nvidia-docker/gpgkey | apt-key add - && curl -s -L https://nvidia.github.io/nvidia-docker/$distribution/nvidia-docker.list 
| tee /etc/apt/sources.list.d/nvidia-docker.list

#Container’ların GPU kullanabilmesi için container-runtime kurulması
apt-get update && apt-get install -y nvidia-container-runtime

#Driver’ın kurulumunun tamamlandığının test edilmesi
nvidia-smi
